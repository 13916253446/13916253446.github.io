1.每秒60帧函数requestAnimationFrame
requestAnimationFrame是浏览器用于定时循环操作的一个接口，类似于setTimeout，主要用途是按帧对网页进行重绘
//示例
function test(){
requestAnimationFrame(test);
};
requestAnimationFrame(test);
2.显示loading动画 showLoading
showLoading挂载在nativeCacheObj原型上,可选参数canClick，默认为false
canClick：true表示显示loading动画的时候，页面还可触发点击事件
canClick：fale相反
//示例
nativeCacheObj.showLoading(true);

lazyImgBase挂载在nativeCacheObj原型上
//示例
var img=nativeCacheObj.lazyImgBase;
4.隐藏loading动画hideLoading
hideLoading挂载在nativeCacheObj原型上

showMask挂载在nativeCacheObj原型上,可选参数callBack表示点击遮罩层的回调函数

//示例

6.隐藏遮罩层hideMask
hideMask挂载在nativeCacheObj原型上

7.自动消失的消息提示showTextMessagebox
showTextMessagebox(msg,delayTime)挂载在nativeCacheObj原型上,必传参数msg表示要显示的消息内容，可选参数delayTime表示经过多少秒之后提示内容消失,默认是800ms
//示例

8.显示成功提示的弹窗showSuccessPopup
showSuccessPopup(option)挂载在nativeCacheObj原型上,必传参数option,类型为Object,字段有Title表示弹窗的标题，Content表示弹窗的正文,Callback表示点击确定后的回调，类型必须为Function
//示例
nativeCacheObj.showSuccessPopup({
Title:"测试",
Content:"数据保存成功了哦",
Callback:function(){
//do something....
}
});

9.获取链接参数的值getUrlParam
getUrlParam(name)挂载在window对象上,必传参数name表示要获取的链接的key
//示例,https://www.taobao.com/?time=1
var time=getUrlParam("time");//1
10.get请求数据 get

//示例
zd.get("www.taobao.com", {time:1}, function(){
//do something....
});

//示例
zd.post("www.taobao.com", {time:1}, function(){
//do something....
});
12.加载文件 loadFile
loadFile(url,option)挂载在zd原型上,必传参数url表示文件的路径，可选参数option类型为Object,里面的key有needInsert表示是否直接插入当前环境默认为true,async表示加载文件是否是异步，默认为false,cache表示是否缓存起来，这里的缓存是永久缓存，如果传入的值为true的话，下次再次请求会先判断缓存里面有没有，如果有直接从缓存拿取，默认为false,alias表示别名也就是缓存里面的key,version表示版本，如果拉取的版本与缓存里面的版本不同的话，会重新发出请求并缓存最新的数据，callBack表示加载文件成功之后的回调
//示例
zd.loadFile("./js/jquery.js",{
needInsert: false,
async: true,
callBack: function (data)//data：文件的内容 {
//do something....
}
});
13.存储以及读取数据
get(key,callBack)挂载在yyq_storage原型上用于读取缓存在数据库(yyq_db)的数据,必传参数key表示要读取数据对应的字段以及callBack表示读取成功之后的回调，会返回存储的数据

set(key,val,callBack)挂载在yyq_storage原型上用于写入数据到数据库(yyq_db)的数据,必传参数key表示数据要写入的字段以及val表示要写的数据，val可以是String类型也可以是Array类型也可以是Object类型以及callBack表示写入之后的回调，会返回一个包含isSucces以及msg字段的对象，表示是否写入成功
//示例,存储数据
yyq_storage.set("yyq_hospitl_tempt",[
{
OrganName:"苏州市立医院",
OrganId:4267
},
{
OrganName:"苏州明日医院",
OrganId:4268
}
],function(MSG){
if(MSG.isSuccess){
}else{
}
});
//示例,拉取数据
yyq_storage.get("yyq_hospitl_tempt",function(data){
if(data!=null&&data!=""){
}
});
